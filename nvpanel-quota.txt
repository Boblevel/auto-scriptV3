#!/bin/bash
# ============================================================
#  nvpanel · quota de bande passante par compte
#  Comptabilise le trafic montant (iptables owner match, par UID)
#  et verrouille le compte au dépassement du quota.
#  Base quota : /etc/nvpanel/db/quota   →  utilisateur  octets_max
#  Compteur   : /etc/nvpanel/db/quota_used
#  Note : mesure le trafic montant généré par les process du compte.
#         (up+down complet = évolution prévue, marquage de paquets)
# ============================================================
QDB=/etc/nvpanel/db/quota
USED=/etc/nvpanel/db/quota_used
mkdir -p /etc/nvpanel/db; touch "$QDB" "$USED"
CHAIN=NVPANEL_QUOTA

action="$1"

ensure_chain() {
  iptables -L "$CHAIN" -n >/dev/null 2>&1 || {
    iptables -N "$CHAIN"
    iptables -I OUTPUT -j "$CHAIN"
  }
}

# add <user> : crée la règle de comptage pour l'UID du compte
if [ "$action" = "add" ]; then
  ensure_chain
  uid=$(id -u "$2" 2>/dev/null) || exit 0
  iptables -C "$CHAIN" -m owner --uid-owner "$uid" -j RETURN 2>/dev/null || \
    iptables -A "$CHAIN" -m owner --uid-owner "$uid" -j RETURN
  exit 0
fi

# check : additionne les compteurs, applique les quotas (défaut : appelé par cron)
ensure_chain
awk '{print $1, $2}' "$QDB" | while read -r user max; do
  [ -z "$user" ] && continue
  uid=$(id -u "$user" 2>/dev/null) || continue
  bytes=$(iptables -L "$CHAIN" -vxn 2>/dev/null | awk -v id="$uid" '$0 ~ ("owner UID match "id"$"){print $2}')
  bytes=${bytes:-0}
  # cumul persistant (les compteurs iptables se remettent à zéro au reboot)
  prev=$(grep "^$user " "$USED" | awk '{print $2}'); prev=${prev:-0}
  total=$(( prev + bytes ))
  grep -q "^$user " "$USED" && sed -i "s/^$user .*/$user $total/" "$USED" || echo "$user $total" >> "$USED"
  if [ "$total" -ge "$max" ] 2>/dev/null; then
    passwd -l "$user" >/dev/null 2>&1     # verrouille au dépassement
  fi
done
