#!/bin/bash
# ============================================================
#  nvpanel · gestion Shadowsocks   (menu-ss)
#  Interface — logique dans nvpanel-cli (proto = ss)
# ============================================================
source /etc/nvpanel/lib/ui.sh
[ "$EUID" -ne 0 ] && { err "Lance en root."; exit 1; }

header(){ banner; center " GESTION SHADOWSOCKS " "${BOLD}${WHT}"; line; }

ensure_xray() {
  if [ ! -f /usr/local/etc/xray/config.json ] || ! command -v xray >/dev/null 2>&1; then
    warn "Xray n'est pas installé."
    read -rp " L'installer maintenant ? (o/n) : " r
    [ "$r" = "o" ] && install-xray || return 1
  fi
}

add_client() {
  header; read -rp " Nom du client : " name
  [ -z "$name" ] && { err "Nom vide"; pause; return; }
  read -rp " Durée (jours) : " days
  out=$(nvpanel-cli xadd ss "$name" "$days")
  clear; header
  if echo "$out" | head -1 | grep -q '^OK'; then
    ok "Client Shadowsocks créé (expire $(echo "$out"|head -1|awk '{print $3}'))"
    echo; printf "   ${GRY}Lien :${NC}\n   ${CYN}%s${NC}\n" "$(echo "$out"|tail -1)"
  else err "Échec : $(echo "$out"|head -1)"; fi
  pause
}
list_clients() {
  header; printf " ${GRY}%-20s %s${NC}\n" "Nom" "Expire"; line
  nvpanel-cli xlist ss | while IFS='|' read -r n e; do printf " %-20s %s\n" "$n" "$e"; done
  pause
}
del_client() {
  header; read -rp " Nom à supprimer : " name
  r=$(nvpanel-cli xdel ss "$name"); echo "$r"|grep -q '^OK' && ok "$name supprimé" || err "$r"; pause
}
renew_client() {
  header; read -rp " Nom : " name; read -rp " Ajouter (jours) : " days
  r=$(nvpanel-cli xrenew ss "$name" "$days")
  echo "$r"|grep -q '^OK' && ok "Prolongé" || err "$r"; pause
}

ensure_xray || exit 0
while true; do
  header
  entry "1" "Ajouter un client"; entry "2" "Lister les clients"
  entry "3" "Prolonger un client"; entry "4" "Supprimer un client"; entry "0" "Retour"
  line
  read -rp "$(printf " ${WHT}Choix :${NC} ")" o
  case "$o" in 1) add_client ;; 2) list_clients ;; 3) renew_client ;; 4) del_client ;; 0) break ;; *) warn "Invalide"; sleep 1 ;; esac
done
