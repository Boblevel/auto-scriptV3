#!/bin/bash
# ============================================================
#  nvpanel · installation SlowDNS (dnstt)  → tunnel DNS vers SSH
#  commande : install-slowdns
# ============================================================
source /etc/nvpanel/lib/ui.sh
[ "$EUID" -ne 0 ] && { err "Lance en root."; exit 1; }

# Binaire dnstt-server : héberge le tien et mets l'URL ici
# (c'est l'équivalent de tes liens raw.githubusercontent NevermoreSSH)
SLDNS_BIN_URL="${SLDNS_BIN_URL:-https://raw.githubusercontent.com/Boblevel/auto-scriptV3/main/dns-server}"
DIR=/etc/nvpanel/slowdns

banner; center " INSTALLATION SLOWDNS " "${BOLD}${WHT}"; line

read -rp " Nameserver (ex. ns.tondomaine.com) : " NS
[ -z "$NS" ] && { err "Nameserver requis"; exit 1; }

mkdir -p "$DIR"; cd "$DIR"

# 1) Binaire
if [ ! -f "$DIR/dns-server" ]; then
  say(){ printf "${CYN}➜${NC} %s\n" "$1"; }
  printf "${CYN}➜${NC} Téléchargement du binaire dnstt…\n"
  curl -sL "$SLDNS_BIN_URL" -o "$DIR/dns-server"
  chmod +x "$DIR/dns-server"
fi

# 2) Clés
if [ ! -f "$DIR/server.key" ]; then
  "$DIR/dns-server" -gen-key -privkey-file "$DIR/server.key" -pubkey-file "$DIR/server.pub"
fi
PUB=$(cat "$DIR/server.pub")
echo "$NS" > "$DIR/ns"

# 3) Redirection UDP 53 → 5300
iptables -I INPUT -p udp --dport 5300 -j ACCEPT
iptables -t nat -I PREROUTING -p udp --dport 53 -j REDIRECT --to-ports 5300
mkdir -p /etc/iptables && iptables-save > /etc/iptables/rules.v4 2>/dev/null || true

# 4) Service (forward vers le SSH local sur 22)
cat > /etc/systemd/system/nvpanel-slowdns.service <<EOF
[Unit]
Description=nvpanel SlowDNS (dnstt)
After=network.target

[Service]
ExecStart=$DIR/dns-server -udp :5300 -privkey-file $DIR/server.key $NS 127.0.0.1:22
Restart=always
User=root

[Install]
WantedBy=multi-user.target
EOF

systemctl daemon-reload
systemctl enable --now nvpanel-slowdns >/dev/null 2>&1

clear; banner; center " SLOWDNS PRÊT " "${BOLD}${GRN}"; line
ok "Service actif."
echo
printf "   ${GRY}À configurer chez ton registrar DNS :${NC}\n"
printf "   • %s        A    → %s\n" "$NS" "$(cat /etc/nvpanel/ip)"
printf "   • (sous-domaine)  NS   → %s\n" "$NS"
echo
printf "   ${GRY}Clé publique (à donner au client) :${NC}\n"
printf "   ${WHT}%s${NC}\n" "$PUB"
pause
