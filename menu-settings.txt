#!/bin/bash
# ============================================================
#  nvpanel · réglages du serveur   (commande : menu-settings)
# ============================================================
source /etc/nvpanel/lib/ui.sh
[ "$EUID" -ne 0 ] && { err "Lance en root."; exit 1; }

header(){ banner; center " RÉGLAGES DU SERVEUR " "${BOLD}${WHT}"; line; }

while true; do
  header
  printf " ${GRY}Domaine :${NC} %s\n" "$(cat /etc/nvpanel/domain 2>/dev/null || echo 'non défini')"
  printf " ${GRY}TLS     :${NC} %s\n" "$([ -f /etc/nvpanel/tls_on ] && printf "${GRN}actif${NC}" || printf "${RED}inactif${NC}")"
  printf " ${GRY}IP      :${NC} %s\n" "$(cat /etc/nvpanel/ip 2>/dev/null)"
  line
  entry "1" "Définir / changer le domaine"
  entry "2" "Activer le TLS (HTTPS)"
  entry "3" "Rafraîchir l'IP publique"
  entry "0" "Retour"
  line
  read -rp "$(printf " ${WHT}Choix :${NC} ")" o
  case "$o" in
    1)
      read -rp " Nouveau domaine (ex. vpn.mondomaine.com) : " d
      [ -n "$d" ] && { echo "$d" > /etc/nvpanel/domain; ok "Domaine enregistré : $d"; } || warn "Inchangé"
      pause ;;
    2)
      if [ -s /etc/nvpanel/domain ]; then install-tls
      else err "Définis d'abord un domaine (option 1)."; pause; fi ;;
    3)
      curl -s ipv4.icanhazip.com > /etc/nvpanel/ip && ok "IP mise à jour : $(cat /etc/nvpanel/ip)"; pause ;;
    0) break ;;
    *) warn "Invalide"; sleep 1 ;;
  esac
done
