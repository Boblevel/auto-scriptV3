#!/bin/bash
# ============================================================
#  nvpanel · nettoyage automatique des comptes expirés
#  (appelé par cron) — SSH + clients Xray/Shadowsocks
# ============================================================
now=$(date +%s)
XCONF=/usr/local/etc/xray/config.json
changed=0

# --- Comptes SSH --------------------------------------------
awk -F: '$3>=1000 && $3<60000 {print $1}' /etc/passwd | while read -r u; do
  exp=$(chage -l "$u" 2>/dev/null | awk -F: '/Account expires/{print $2}' | xargs)
  [ "$exp" = "never" ] || [ -z "$exp" ] && continue
  es=$(date -d "$exp" +%s 2>/dev/null) || continue
  if [ "$now" -ge "$es" ]; then
    userdel -f "$u" 2>/dev/null
    sed -i "/^### $u /d" /etc/nvpanel/db/ssh 2>/dev/null
  fi
done

# --- Clients Xray / Shadowsocks -----------------------------
purge_proto() {
  local db="$1" tag="$2"
  [ -f "$db" ] || return
  while read -r _ name uuid exp; do
    [ -z "$name" ] && continue
    es=$(date -d "$exp" +%s 2>/dev/null) || continue
    if [ "$now" -ge "$es" ]; then
      if [ -f "$XCONF" ]; then
        jq "(.inbounds[] | select(.tag==\"$tag\") | .settings.clients) |= map(select(.email!=\"$name\"))" \
          "$XCONF" > "$XCONF.tmp" && mv "$XCONF.tmp" "$XCONF"
        changed=1
      fi
      sed -i "/^### $name /d" "$db"
    fi
  done < "$db"
}

purge_proto /etc/nvpanel/db/vmess       vmess-in
purge_proto /etc/nvpanel/db/vless       vless-in
purge_proto /etc/nvpanel/db/trojan      trojan-in
purge_proto /etc/nvpanel/db/shadowsocks ss-in

[ "$changed" = "1" ] && systemctl restart xray >/dev/null 2>&1
exit 0
